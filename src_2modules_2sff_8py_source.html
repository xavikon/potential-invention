192 def _detect_capabilities(self) -> None:
193 """
194 Detect which optional capabilities are supported by the module.
195 Updates the _supported_capabilities set.
196 """
197 # Start with required capabilities
198 self._supported_capabilities_supported_capabilities = self._required_capabilities.copy()
199
200 try:
201 # Read diagnostic monitoring type
202 diag_type = self.hw.read_register(0x5C) # SFF-8472 diagnostic monitoring type
203
204 # Check for optional monitoring capabilities
205 if diag_type & 0x40: # Digital diagnostic monitoring implemented
206 self._supported_capabilities_supported_capabilities.update({
207 ModuleCapability.TX_BIAS_MONITORING,
208 ModuleCapability.TX_POWER_MONITORING,
209 ModuleCapability.RX_POWER_MONITORING
210 })
211
212 if diag_type & 0x08: # External calibration
213 self._supported_capabilities_supported_capabilities.add(ModuleCapability.ALARM_THRESHOLDS)
214
215 # Check for optional control capabilities
216 control_bits = self.hw.read_register(0x6E) # Enhanced options
217 if control_bits & 0x40: # TX_DISABLE implemented
218 self._supported_capabilities_supported_capabilities.add(ModuleCapability.TX_DISABLE)
219 if control_bits & 0x20: # TX_FAULT implemented
220 self._supported_capabilities_supported_capabilities.add(ModuleCapability.TX_FAULT)
221 if control_bits & 0x10: # RX_LOS implemented
222 self._supported_capabilities_supported_capabilities.add(ModuleCapability.RX_LOS)
223
224 except Exception as e:
225 # Log warning but continue - we'll work with just required capabilities
226 print(f"Warning: Error detecting optional capabilities: {str(e)}")

228 def _read_string(self, start_address: int, length: int) -> str:
229 """Read a string from consecutive memory addresses"""
230 bytes_data = []
231 for offset in range(length):
232 try:
233 byte = self.hw.read_register(start_address + offset)
234 bytes_data.append(byte)
235 except Exception:
236 break
237
238 return "".join(chr(b) for b in bytes_data if 32 <= b <= 126)

240 def _read_word(self, address: int) -> int:
241 """Read a 16-bit word from memory"""
242 msb = self.hw.read_register(address)
243 lsb = self.hw.read_register(address + 1)
