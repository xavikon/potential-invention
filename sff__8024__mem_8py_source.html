93# --- Base Memory Space Class ---

95class MemorySpace:
96 """
97 Base class for a memory space. Subclasses define specific layouts.
98 """
99 def __init__(self, data: bytes):
100 """
101 Initializes the memory space with raw EEPROM data.
102 :param data: A bytes object representing the memory space.
103 """
104 self._data = data
105 self._fields = {} # To store decoded fields
106
107 def get_byte(self, address: int) -> int:
108 """Retrieves a single byte from the memory space."""
109 if 0 <= address < len(self._data):
110 return self._data[address]
111 raise IndexError(f"Address 0x{address:02X} out of bounds.")
112
113 def get_bytes(self, start_address: int, length: int) -> bytes:
114 """Retrieves a sequence of bytes from the memory space."""
115 end_address = start_address + length
116 if 0 <= start_address < end_address <= len(self._data):
117 return self._data[start_address:end_address]
118 raise IndexError(f"Address range 0x{start_address:02X}-0x{end_address-1:02X} out of bounds.")
119
120 def get_string(self, start_address: int, length: int) -> str:
121 """Retrieves a string from the memory space, stripping nulls."""
122 return self.get_bytes(start_address, length).decode('ascii').strip('\0')
123
124 def _decode_field(self, name, start_addr, length, decoder=None):
125 """Helper to decode and store a field."""
126 raw_data = self.get_bytes(start_addr, length)
127 if decoder:
128 decoded_value = decoder(raw_data if length > 1 else raw_data[0])
129 else:
130 decoded_value = raw_data
131 self._fields[name] = decoded_value
132 return decoded_value
133
134 def __repr__(self):
135 return f"{self.__class__.__name__}(data_len={len(self._data)})"
136
137 def to_dict(self):
138 """Returns a dictionary representation of the decoded fields."""
139 return self._fields
