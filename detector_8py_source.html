51 def identify_module_type(self) -> ModuleType:
52 """
53 Identify the type of module that is present.
54 This is done by reading the identifier bytes and checking
55 the module's response pattern.
56
57 Returns:
58 The detected module type
59
60 Raises:
61 NoModuleError: If no module is present
62 """
63 if not self.is_module_present():
64 raise NoModuleError("No module detected")
65
66 # Try reading CMIS identifier first
67 try:
68 cmis_id = self.hw.read_register(CMISRegisters.IDENTIFIER.offset)
69 if cmis_id in [0x0D, 0x11]: # Common CMIS identifier values
70 return ModuleType.CMIS
71 except Exception:
72 pass # Continue to SFF check if CMIS read fails
73
74 # Try reading SFF identifier
75 try:
76 sff_id = self.hw.read_register(SFFRegisters.IDENTIFIER.offset)
77 if sff_id in [0x03, 0x0C]: # Common SFF identifier values
78 return ModuleType.SFF
79 except Exception:
80 pass
81
82 return ModuleType.UNKNOWN

141 def _read_string(self, start_address: int, length: int) -> str:
142 """
143 Read a string from consecutive memory addresses.
144
145 Args:
146 start_address: Starting memory address
147 length: Number of bytes to read
148
149 Returns:
150 The decoded string, with non-printable characters removed
151 """
152 bytes_data = []
153 for offset in range(length):
154 try:
155 byte = self.hw.read_register(start_address + offset)
156 bytes_data.append(byte)
