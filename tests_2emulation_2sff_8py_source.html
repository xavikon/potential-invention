89 def _initialize_memory_map(self) -> None:
90 """!
91 Initialize the SFF memory map.
92
93 This method sets up the complete memory map for an SFF module including:
94 - A0h page: Basic identification and capability info
95 - A2h page: Diagnostic monitoring
96
97 Memory map initialization includes:
98 1. Creating required memory pages
99 2. Writing identification information
100 3. Setting up diagnostic monitoring capabilities
101 4. Configuring alarm thresholds
102 5. Initializing monitoring values
103
104 The memory map follows either SFF-8472 (for SFP) or SFF-8636 (for QSFP)
105 depending on the module configuration.
106
107 @note This is called automatically by the constructor and during reset.
108 """
109 # Initialize pages
110 self.add_memory_pages([0x00, 0x01, 0xA0, 0xA2]) # Add standard and diagnostic pages
111
112 # Clear all memory pages first
113 for page in [0x00, 0x01, 0xA0, 0xA2]:
114 self.memory_map.select_page(page)
115 for addr in range(256):
116 self.memory_map.write_byte(addr, 0x00)
117
118 # Select A0 page for writing basic information
119 self.memory_map.select_page(0xA0)
120 # Basic module information
121 self.memory_map.write_byte(0x00, self.config.identifier) # Identifier
122 self._write_string(0x14, self.config.vendor_name, 16)
123 self._write_string(0x28, self.config.part_number, 16)
124 self._write_string(0x38, self.config.revision, 4)
125 self._write_string(0x44, self.config.serial_number, 16)
126
127 # Switch to A2 page for diagnostics setup
128 self.memory_map.select_page(0xA2)
129
130 # Enable digital diagnostics
131 self.memory_map.write_byte(0x5C, 0x40) # Digital diagnostics implemented
132
133 # Initial monitoring values
134 self.update_monitoring()
135
136 # Basic module information
137 self.memory_map.write_byte(0x00, self.config.identifier) # Identifier
138 self.memory_map.write_byte(0x01, 0x00) # Status
139 self._write_string(0x14, self.config.vendor_name, 16)
140 self._write_string(0x28, self.config.part_number, 16)
141 self._write_string(0x38, self.config.revision, 4)
142 self._write_string(0x44, self.config.serial_number, 16)
143
144 # Module characteristics
145 self.memory_map.write_byte(0x3C, int(self.config.nominal_bit_rate))
146
147 if self.config.length_meters is not None:
148 self.memory_map.write_byte(0x12, int(self.config.length_meters))
149 if self.config.wavelength_nm is not None:
150 wavelength = int(self.config.wavelength_nm / 20) # Convert to 20nm units
151 self.memory_map.write_word(0x3C, wavelength)
152
153 # Select A2 page for diagnostics
154 self.memory_map.select_page(0xA2)
155
156 # Diagnostic capabilities
157 self.memory_map.write_byte(0x5C, 0x40) # Digital diagnostics implemented
158
159 # Alarm thresholds (example values)
160 self.memory_map.write_word(0x00, self._encode_temperature(75.0)) # Temp high alarm
161 self.memory_map.write_word(0x02, self._encode_temperature(-5.0)) # Temp low alarm
162 self.memory_map.write_word(0x04, self._encode_voltage(3.6)) # Voltage high alarm
163 self.memory_map.write_word(0x06, self._encode_voltage(3.0)) # Voltage low alarm
164
165 # Update initial monitoring values

168 def update_monitoring(self) -> None:
169 """!
170 Update monitoring values in memory map.
171
172 Updates all diagnostic monitoring values in the A2h page including:
173 - Temperature (bytes 96-97)
174 - Voltage (bytes 98-99)
175 - TX bias current (bytes 100-101)
176 - TX output power (bytes 102-103)
177 - RX input power (bytes 104-105)
178
179 Each value is updated with small random variations to simulate
180 real-world behavior. The variation ranges are:
181 - Temperature: ±0.05°C
182 - Voltage: ±0.01V
183 - Bias current: ±0.2mA
184 - Optical power: ±0.01mW
185
186 The method also updates:
187 - Status bits (byte 110)
188 - Alarm flags (byte 112)
189
190 @note The current page selection is preserved across the update.
191 """
192 # Add some random variation to values (reduced for stability)
193 temp_variation = random.uniform(-0.05, 0.05)
194 voltage_variation = random.uniform(-0.01, 0.01)
195
196 # Get current page
197 current_page = self.memory_map.current_page
198
199 # Select A2h page for diagnostics
200 self.memory_map.select_page(0xA2)
201
202 # Temperature and voltage (mandatory)
203 temp = self._temperature + temp_variation
204 self.memory_map.write_word(0x60, self._encode_temperature(temp))
205 self.memory_map.write_word(0x62, self._encode_voltage(self._voltage + voltage_variation))
206
207 # Optional monitoring values for optical modules
208 if not self.config.media_type.name.startswith('COPPER'):
209 bias_variation = random.uniform(-0.2, 0.2)
210 power_variation = random.uniform(-0.01, 0.01)
211
212 self.memory_map.write_word(0x64, self._encode_bias(self._tx_bias_ma + bias_variation))
213 self.memory_map.write_word(0x66, self._encode_power(self._tx_power_mw + power_variation))
214 self.memory_map.write_word(0x68, self._encode_power(self._rx_power_mw + power_variation))
215
216 # Update status and alarm flags
217 status = 0x00
218 alarms = 0x00
219
220 # Status bits
221 if self._tx_disable:
222 status |= 0x40
223 if self._tx_fault:
224 status |= 0x20
225 if self._rx_los:
226 status |= 0x10
227
228 # Alarm bits
229 if temp > 75.0: # High temp alarm
230 alarms |= 0x80
231 if temp < -5.0: # Low temp alarm
232 alarms |= 0x40
233 if self._voltage > 3.6: # High voltage alarm
234 alarms |= 0x20
235 if self._voltage < 3.0: # Low voltage alarm
236 alarms |= 0x10
237
238 self.memory_map.write_byte(0x6E, status)
239 self.memory_map.write_byte(0x70, alarms)
240
241 # Restore original page
242 self.memory_map.select_page(current_page)

244 def set_tx_disable(self, disable: bool) -> None:
245 """!
246 Set TX disable state.
247
248 @param disable True to disable TX output, False to enable
249
250 Controls the transmitter disable state:
251 - When disabled (True):
252 - TX power drops to 0
253 - TX disable bit (bit 6) is set in status register
254 - Power registers are updated immediately
255 - When enabled (False):
256 - TX power returns to normal
257 - TX disable bit is cleared
258 - Normal monitoring resumes
259
260 The TX disable state is reflected in:
261 - Status byte (110): bit 6
262 - TX power registers (102-103)
263
264 Example:
265 @code
266 module.set_tx_disable(True) # Disable TX
267 status = hardware.read_register(0xA2, 110)
268 assert status & 0x40 # Check TX disable bit
269
270 module.set_tx_disable(False) # Enable TX
271 status = hardware.read_register(0xA2, 110)
272 assert not (status & 0x40) # Bit should be cleared
273 @endcode
274
275 @note The current page selection is preserved.
276 """
277 # Save current page
278 current_page = self.memory_map.current_page
279
280 self._tx_disable = disable
281 if disable:
282 self._tx_power_mw = 0.0
283 # Force update with exactly zero power and skip random variation
284 self.memory_map.select_page(0xA2)
285 self.memory_map.write_word(0x66, 0)
286 # Set TX disable bit
287 status = self.memory_map.read_byte(0x6E)
288 status |= 0x40
289 self.memory_map.write_byte(0x6E, status)
290 else:
291 self._tx_power_mw = 0.5 # Return to typical power
292 # Update TX power value
293 self.memory_map.select_page(0xA2)
294 power_value = self._encode_power(self._tx_power_mw)
295 self.memory_map.write_word(0x66, power_value)
296 # Clear TX disable bit
297 status = self.memory_map.read_byte(0x6E)
298 status &= ~0x40
299 self.memory_map.write_byte(0x6E, status)
300
301 # Restore original page
302 self.memory_map.select_page(current_page)

304 def simulate_fault(self, fault_type: str, state: bool) -> None:
305 """!
306 Simulate various fault conditions.
307
308 @param fault_type Type of fault to simulate ('tx_fault' or 'rx_los')
309 @param state True to activate the fault, False to clear it
310 @throws EmulationError if fault_type is unknown
311
312 Supported fault types:
313 - 'tx_fault': Transmitter fault
314 - Sets/clears bit 5 in status register
315 - Forces TX power to 0 when active
316 - 'rx_los': Receiver loss of signal
317 - Sets/clears bit 4 in status register
318 - Forces RX power to 0 when active
319
320 For each fault type:
321 1. Updates internal state
322 2. Modifies status register
323 3. Updates affected monitoring values
324 4. Preserves page selection
325
326 Example:
327 @code
328 # Simulate TX fault
329 module.simulate_fault('tx_fault', True)
330 status = hardware.read_register(0xA2, 110)
331 assert status & 0x20 # Check TX fault bit
332
333 # Clear TX fault
334 module.simulate_fault('tx_fault', False)
335 status = hardware.read_register(0xA2, 110)
336 assert not (status & 0x20)
337 @endcode
338 """
339 # Save current page
340 current_page = self.memory_map.current_page
341
342 if fault_type == 'tx_fault':
343 self._tx_fault = state
344 if state:
345 self._tx_power_mw = 0.0
346 # Force update with exactly zero power and skip random variation
347 self.memory_map.select_page(0xA2)
348 self.memory_map.write_word(0x66, 0)
349 # Update status bits
350 status = self.memory_map.read_byte(0x6E)
351 status |= 0x20 # Set TX fault bit
352 self.memory_map.write_byte(0x6E, status)
353 else:
354 self._tx_power_mw = 0.5
355 # Clear TX fault bit
356 self.memory_map.select_page(0xA2)
357 status = self.memory_map.read_byte(0x6E)
358 status &= ~0x20
359 self.memory_map.write_byte(0x6E, status)
360 elif fault_type == 'rx_los':
361 self._rx_los = state
362 if state:
363 self._rx_power_mw = 0.0
364 # Force update with exactly zero power and skip random variation
365 self.memory_map.select_page(0xA2)
366 self.memory_map.write_word(0x68, 0)
367 # Update status bits
368 status = self.memory_map.read_byte(0x6E)
369 status |= 0x10 # Set RX LOS bit
370 self.memory_map.write_byte(0x6E, status)
371 else:
372 self._rx_power_mw = 0.4
373 # Clear RX LOS bit
374 self.memory_map.select_page(0xA2)
375 status = self.memory_map.read_byte(0x6E)
376 status &= ~0x10
377 self.memory_map.write_byte(0x6E, status)
378 else:
379 raise EmulationError(f"Unknown fault type: {fault_type}")
380
381 # Restore original page
382 self.memory_map.select_page(current_page)
