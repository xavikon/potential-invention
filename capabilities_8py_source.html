90 def validate_module(self, module: BaseModule) -> Dict[str, List[ModuleCapability]]:
91 """
92 Validate a module's capabilities against requirements.
93
94 Args:
95 module: The module to validate
96
97 Returns:
98 Dictionary containing:
99 - 'missing_required': List of required capabilities that are missing
100 - 'supported_optional': List of optional capabilities that are supported
101 - 'unsupported_optional': List of optional capabilities that are not supported
102 """
103 module_type = module.get_identification().type
104 supported = module.get_capabilities()
105
106 required = self.get_required_capabilities(module_type)
107 optional = self.get_optional_capabilities(module_type)
108
109 missing_required = [cap for cap in required if cap not in supported]
110 supported_optional = [cap for cap in optional if cap in supported]
111 unsupported_optional = [cap for cap in optional if cap not in supported]
112
113 return {
114 'missing_required': missing_required,
115 'supported_optional': supported_optional,
116 'unsupported_optional': unsupported_optional
117 }
