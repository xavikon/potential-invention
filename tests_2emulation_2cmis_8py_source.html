67 def _initialize_memory_map(self) -> None:
68 """Initialize the CMIS memory map"""
69 # Lower Memory Page (Page 0)
70 self.memory_map.write_byte(0x00, self.config.identifier) # Identifier
71 self._write_string(0x10, self.config.vendor_name, 16)
72 self._write_string(0x20, self.config.part_number, 16)
73 self._write_string(0x30, self.config.revision, 2)
74 self._write_string(0x40, self.config.serial_number, 16)
75
76 # Status and Control Pages
77 for page in [0x80, 0x81, 0x82, 0x83, 0x84, 0x85]:
78 self.memory_map.add_page(page)
79

81 self.memory_map.select_page(0x80)
82 features = 0x00
83 if not self.config.media_type.name.startswith('COPPER'):
84 features |= 0x07 # Power monitoring supported
85 if len(self.config.supported_rates) > 1:
86 features |= 0x20 # Programmable rates supported
87 self.memory_map.write_byte(0x02, features)

89 # Application Advertisement (Page 0x10)
90 self.memory_map.add_page(0x10)

92
93 # Write supported rates as application codes
94 for i, rate in enumerate(self.config.supported_rates):
95 app_code = self._encode_application(rate)
96 self.memory_map.write_word(0x20 + (i * 2), app_code)

188 def simulate_fault(self, channel: int, fault_type: str, state: bool) -> None:
