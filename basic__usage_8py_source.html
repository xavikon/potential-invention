39def print_capability_info(module: BaseModule, capability_mgr: CapabilityManager) -> None:
40 """Print capability information for a module"""
41 validation = capability_mgr.validate_module(module)
42 print(f"\nCapability Information:")
43
44 if validation['missing_required']:
45 print(" Missing Required Capabilities:")
46 for cap in validation['missing_required']:
47 print(f" - {capability_mgr.describe_capability(cap)}")
48
49 print("\n Supported Optional Capabilities:")
50 for cap in validation['supported_optional']:
51 verified = capability_mgr.verify_capability(module, cap)
52 status = "Functional" if verified else "Non-functional"
53 print(f" - {capability_mgr.describe_capability(cap)} ({status})")
54
55 print("\n Unsupported Optional Capabilities:")
56 for cap in validation['unsupported_optional']:
