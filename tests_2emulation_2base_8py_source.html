127class MemoryMap:
128 """!
129 Represents a module's memory map.
130
131 The MemoryMap class provides a complete emulation of a module's register-based
132 memory interface, including:
133 - Multiple memory pages (like A0h and A2h in SFF modules)
134 - Page selection mechanism
135 - Byte and word access methods
136 - Bounds checking and error handling
137
138 Example usage:
139 @code
140 memory = MemoryMap(size=256) # Create 256-byte pages
141 memory.add_page(0xA0) # Add A0h page
142 memory.add_page(0xA2) # Add A2h page
143
144 memory.select_page(0xA0) # Select A0h page
145 memory.write_byte(0, 0x03) # Write identifier
146
147 memory.select_page(0xA2) # Switch to A2h page
148 temp = memory.read_word(96) # Read temperature register
149 @endcode
150 """
151
152 def __init__(self, size: int = 256):
153 """!
154 Initialize memory map.
155
156 @param size Size of each memory page in bytes (default: 256)
157
158 Creates a new memory map with an initial page 0. Additional pages
159 can be added using the add_page() method.
160 """
161 ## Dictionary mapping page numbers to byte arrays
162 self._pages: Dict[int, bytearray] = {0: bytearray(size)}
163 ## Currently selected memory page
164 self._current_page = 0
165 ## Size of each memory page in bytes
166 self._size = size
167
168 def add_page(self, page_number: int) -> None:
169 """!
170 Add a new memory page to the map.
171
172 @param page_number The page number to add (e.g., 0xA0, 0xA2)
173
174 Creates a new page initialized to all zeros. If the page already exists,
175 this method has no effect.
176 """
177 if page_number not in self._pages:
178 self._pages[page_number] = bytearray(self._size)
179
180 @property
181 def current_page(self) -> int:
182 """!
183 Get the currently selected page number.
184
185 @return The page number that is currently selected for access
186 """
187 return self._current_page
188
189 def select_page(self, page: int) -> None:
190 """!
191 Select a memory page for subsequent access.
192
193 @param page The page number to select
194 @throws EmulationError if the page does not exist
195
196 All subsequent read and write operations will access the selected page
197 until a different page is selected.
198 """
199 if page not in self._pages:
200 raise EmulationError(f"Page {page} does not exist")
201 self._current_page = page
202
203 def read_byte(self, address: int) -> int:
204 """!
205 Read a byte from the current page.
206
207 @param address Memory address to read (0-255)
208 @return The byte value at the specified address
209 @throws EmulationError if address is out of range
210
211 Reads a single byte from the specified address in the currently
212 selected memory page.
213 """
214 if not 0 <= address < self._size:
215 raise EmulationError(f"Address {address} out of range")
216 return self._pages[self._current_page][address]
217
218 def write_byte(self, address: int, value: int) -> None:
219 """!
220 Write a byte to the current page.
221
222 @param address Memory address to write (0-255)
223 @param value Byte value to write (0-255)
224 @throws EmulationError if address or value is out of range
225
226 Writes a single byte to the specified address in the currently
227 selected memory page.
228 """
229 if not 0 <= address < self._size:
230 raise EmulationError(f"Address {address} out of range")
231 if not 0 <= value <= 255:
232 raise EmulationError(f"Value {value} out of range")
233 self._pages[self._current_page][address] = value
234
235 def write_bytes(self, start_address: int, data: bytes) -> None:
236 """!
237 Write multiple consecutive bytes starting at an address.
238
239 @param start_address Starting memory address
240 @param data Bytes to write
241 @throws EmulationError if the operation would exceed page boundaries
242
243 Writes a sequence of bytes to consecutive addresses starting at
244 the specified address in the currently selected page.
245 """
246 if not 0 <= start_address < self._size:
247 raise EmulationError(f"Start address {start_address} out of range")
248 if start_address + len(data) > self._size:
249 raise EmulationError("Data would exceed page size")
250 for i, value in enumerate(data):
251 self._pages[self._current_page][start_address + i] = value
252
253 def read_word(self, address: int) -> int:
254 """!
255 Read a 16-bit word from the current page.
256
257 @param address Memory address of MSB (must have valid LSB at address+1)
258 @return 16-bit value composed of [address]=MSB, [address+1]=LSB
259 @throws EmulationError if address is out of range
260
261 Reads two consecutive bytes and combines them into a 16-bit word.
262 The byte at the specified address is treated as the MSB.
263 """
264 if not 0 <= address < self._size - 1:
265 raise EmulationError(f"Address {address} out of range for word access")
266 msb = self.read_byte(address)
267 lsb = self.read_byte(address + 1)
268 return (msb << 8) | lsb
269
270 def write_word(self, address: int, value: int) -> None:
271 """!
272 Write a 16-bit word to the current page.
273
274 @param address Memory address for MSB (LSB will be at address+1)
275 @param value 16-bit value to write
276 @throws EmulationError if address or value is out of range
277
278 Splits a 16-bit value into two bytes and writes them to consecutive
279 addresses, with the MSB at the specified address.
280 """
281 if not 0 <= address < self._size - 1:
282 raise EmulationError(f"Address {address} out of range for word access")
283 if not 0 <= value <= 65535:
284 raise EmulationError(f"Value {value} out of range for word")
285 msb = (value >> 8) & 0xFF
286 lsb = value & 0xFF
287 self.write_byte(address, msb)
288 self.write_byte(address + 1, lsb)

290class EmulatedModule(ABC):
291 """!
292 Base class for emulated modules.
293
294 The EmulatedModule class provides the foundation for implementing module
295 emulators. It handles:
296 - Memory map management
297 - GPIO signal states
298 - Basic module status
299 - Monitoring values
300 - Configuration storage
301
302 Derived classes must implement the _initialize_memory_map() and
303 update_monitoring() methods to provide module-specific functionality.
304
305 Example:
306 @code
307 class SFFModule(EmulatedModule):
308 def _initialize_memory_map(self):
309 self.add_memory_pages([0xA0, 0xA2])
310 self.memory_map.select_page(0xA0)
311 self.memory_map.write_byte(0, self.config.identifier)
312 # ... initialize other registers
313
314 def update_monitoring(self):
315 self.memory_map.select_page(0xA2)
316 temp = self._encode_temperature(self._temperature)
317 self.memory_map.write_word(96, temp)
318 # ... update other monitoring values
319 @endcode
320 """
321
322 def __init__(self, config: ModuleConfig):
323 """!
324 Initialize emulated module.
325
326 @param config Configuration parameters for the module
327
328 Sets up the base memory map and initializes all monitoring values
329 to their defaults. Derived classes should call this before their
330 own initialization.
331 """
332 self.config = config
333 self.memory_map = MemoryMap()
334 self._gpio_states = {
335 'mod_present': True,
336 'interrupt': False,
337 'reset': False,
338 'lpmode': False
339 }
340 self._temperature = 25.0 # Default temperature in Celsius
341 self._voltage = 3.3 # Default voltage in V
342
343 # Initialize monitoring values
344 self._tx_disable = False
345 self._tx_fault = False
346 self._rx_los = False
347 self._tx_bias_ma = 0.0
348 self._tx_power_mw = 0.0
349 self._rx_power_mw = 0.0
350
351 # Set default monitoring values based on module type
352 if not config.media_type.name.startswith('COPPER'):
353 self._tx_bias_ma = 30.0 # Typical bias current
354 self._tx_power_mw = 0.5 # Typical TX power
355 self._rx_power_mw = 0.4 # Typical RX power

441 def _encode_temperature(self, temp: float) -> int:
442 """Encode temperature value for memory map"""
443 # Convert to 16-bit signed value in 1/256 degree units
444 # Round to nearest 1/256 to avoid precision loss
445 val = round(temp * 256.0)
446 # Handle negative temperatures correctly
447 if val < 0:
448 val = val & 0xFFFF # Convert to unsigned 16-bit
449 return val & 0xFFFF # Ensure 16-bit unsigned value
450
451 def _encode_voltage(self, voltage: float) -> int:
452 """Encode voltage value for memory map"""
453 return int(voltage * 10000.0) # Convert to units of 100µV
454
455 def _encode_power(self, power: float) -> int:
456 """Encode optical power value for memory map"""
457 # Clamp power to non-negative value
458 power = max(0.0, power)
459 # Convert to units of 0.1µW and ensure 16-bit unsigned
460 return int(power * 10000.0) & 0xFFFF
461
462 def _encode_bias(self, bias: float) -> int:
463 """Encode bias current value for memory map"""
464 return int(bias * 500.0) # Convert to units of 2µA

466 def _write_string(self, start_address: int, string: str, max_length: int) -> None:
467 """Write a string to memory map with null termination"""
468 # Convert string to bytes, limiting length to max_length - 1 to leave room for null termination
469 encoded = string.encode('ascii')[:max_length-1]
470 # Write string bytes
471 self.memory_map.write_bytes(start_address, encoded)
472 # Add null terminator
473 self.memory_map.write_byte(start_address + len(encoded), 0)
474 # Fill remaining space with zeros
475 remaining = max_length - len(encoded) - 1
476 if remaining > 0:
477 self.memory_map.write_bytes(start_address + len(encoded) + 1, b'\x00' * remaining)
